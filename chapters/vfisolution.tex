    \E_{t}[V_{t+1}(M_{t+1}, W_{t+1}, K_{t+1}, Q_{t+1})] = \int_{\epsilon_{t+1}}  \int_{\rho_{t+1}} \int_{\gamma_{t+1}} V_{t+1} (M_{t+1}, W_{t+1}, K_{t+1}, Q_{t+1}) d \gamma_{t+1} d\rho_{t+1}  d \epsilon_{t+1}
